.MODEL SMALL
.STACK 1000H
.DATA
	number DB "00000$"
	g DW 1 DUP (000H)

.CODE
main PROC
	MOV AX, @DATA
	MOV DS, AX
	PUSH BP
	MOV BP, SP
	SUB SP, 12
	PUSH AX
	MOV AX, 1
	POP BX
	PUSH AX
	MOV AX, 2
	MUL BX
	MOV BX, AX
	MOV AX, 12
	SUB AX, BX
	MOV BX, AX
	POP AX
	MOV SI, BX
	NEG SI
	MOV [BP+SI], AX
	ADD SP, 12
	POP BP
	MOV AX,4CH
	INT 21H
main ENDP
print_output PROC NEAR
	PUSH BP
	MOV BP, SP
	MOV BX, 0
PRINT_1:
	MOV DX, 0
	MOV CX, 10
	DIV CX
	PUSH DX
	INC BX
	CMP AX, 0
	JG PRINT_1
PRINT_2:
	POP DX
	ADD DX, 30H
	DEC BX
	MOV AL, 0
	MOV AH, 2
	INT 21H
	CMP BX, 0
	JG PRINT_2
	POP BP
	RET 0
